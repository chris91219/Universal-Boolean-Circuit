#
# Use relative to the UBC-Results root, e.g.
#   rclone copy ~/scratch/UBC-Results GDrive:UofT_PhD/Collab_Prof_Kratsios/UBC-Results --filter-from scripts/results_rclone_filter.txt --progress
#   rclone copy GDrive:UofT_PhD/Collab_Prof_Kratsios/UBC-Results .results/UBC-Results --filter-from scripts/results_rclone_filter.txt --progress
#
# Kept file types under the named roots:
#   *.json       summary.json, config.json, sample.meta.json, dataset.sample.meta.json
#   *.jsonl      results.jsonl
#   *.csv        expr_table.csv, meta_runs.csv, meta_groups.csv, best_*.csv, diagnostics tables
#   *.yaml/*.yml config.used.yaml
#

# campaigns. Model checkpoints (*.pt) and all other files are excluded.
